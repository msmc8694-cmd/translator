body { text-align: center; font-family: Arial; } textarea { width: 80%; height: 100px; }async function translateText() { let text = document.getElementById("inputText").value; let response = await fetch("https://api.mymemory.translated.net/get?q=" + encodeURIComponent(text) + "&langpair=en|ar"); let data = await response.json(); document.getElementById("outputText").value = data.responseData.translatedText; }
